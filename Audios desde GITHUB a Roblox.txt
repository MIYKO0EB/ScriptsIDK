local HttpService = game:GetService("HttpService")

-- The URL to the MP3 file from GitHub (HAS TO BE RAW OR ELSE WONT WORK)
local url = "https://github.com/r3k33551-bot/UMTP/raw/refs/heads/main/esquii%20%20Time%20Paradox.mp3"


-- Save the sound data to a local file
writefile("LoveForYou.mp3", game:HttpGet(url))

print("Sound file saved successfully!")
local sound = Instance.new('Sound')
sound.Parent = game.Players.LocalPlayer.Character
sound.Volume = 1
sound.SoundId = getcustomasset('LoveForYou.mp3')
sound:Play()
