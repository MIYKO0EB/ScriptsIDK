local dummy = workspace.Live:WaitForChild("Weakest Dummy")
local humanoid = dummy:WaitForChild("Humanoid")
local torso = dummy:WaitForChild("Torso")

-- ðŸ”¹ Reproducir animaciÃ³n
local anim = Instance.new("Animation")
anim.AnimationId = "rbxassetid://83905432418191"

track:Play()

-- ðŸ”¹ Clonar y activar VFX
local vfx = game.ReplicatedStorage.Emotes.VFX.Embers:Clone()
vfx.Parent = torso

for _,d in ipairs(vfx:GetDescendants()) do
	if d:IsA("ParticleEmitter") then
		d:Emit(50)
	elseif d:IsA("Beam") or d:IsA("Trail") or d:IsA("Light") then
		d.Enabled = true
	end
end
