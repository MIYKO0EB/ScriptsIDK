--ACOMODAR LAS HABILIDADES PARA JUGAR MAS COMODO

local VirtualInputManager = game:GetService("VirtualInputManager")
local Players = game:GetService("Players")
local player = Players.LocalPlayer

-- Crear ScreenGui
local gui = Instance.new("ScreenGui")
gui.Name = "SkillButtonsGUI"
gui.ResetOnSpawn = false
gui.Parent = player:WaitForChild("PlayerGui")

-- Estado de toggles
local moverActivo = true
local ocultarActivo = false

-- Referencia a la barra de skills original
local PlayerGui = player:WaitForChild("PlayerGui")
local skillBar = PlayerGui:FindFirstChild("Hotbar") or PlayerGui:FindFirstChild("Inventory") or nil

-- 游댳 Funci칩n para crear un bot칩n circular
local function crearBoton(numero, posX)
	local button = Instance.new("TextButton")
	button.Size = UDim2.new(0, 70, 0, 70) -- tama침o redondo
	button.Position = UDim2.new(posX, 0, 0.75, 0)
	button.Text = tostring(numero)
	button.Font = Enum.Font.GothamBold
	button.TextScaled = true
	button.BackgroundColor3 = Color3.fromRGB(40, 40, 40)
	button.TextColor3 = Color3.fromRGB(255, 255, 255)
	button.Parent = gui
	button.Active = true
	button.Draggable = moverActivo -- se puede mover solo si toggle activo

	-- 游댳 Hacerlo circular
	local corner = Instance.new("UICorner")
	corner.CornerRadius = UDim.new(1, 0)
	corner.Parent = button

	-- 游댳 A침adir sombra
	local shadow = Instance.new("ImageLabel")
	shadow.Size = UDim2.new(1.3, 0, 1.3, 0)
	shadow.Position = UDim2.new(-0.15, 0, -0.15, 0)
	shadow.Image = "rbxassetid://1316045217"
	shadow.ImageColor3 = Color3.fromRGB(0, 0, 0)
	shadow.ImageTransparency = 0.5
	shadow.BackgroundTransparency = 1
	shadow.ZIndex = 0
	shadow.Parent = button

	-- 游댳 Acci칩n al presionar (simular tecla)
	button.MouseButton1Click:Connect(function()
		local keyMap = {
			[1] = Enum.KeyCode.One,
			[2] = Enum.KeyCode.Two,
			[3] = Enum.KeyCode.Three,
			[4] = Enum.KeyCode.Four,
		}
		local keycode = keyMap[numero]
		if keycode then
			VirtualInputManager:SendKeyEvent(true, keycode, false, game)
			task.wait(0.1)
			VirtualInputManager:SendKeyEvent(false, keycode, false, game)
		end
	end)

	return button
end

-- Guardar referencia a los 4 botones
local botones = {
	crearBoton(1, 0.25),
	crearBoton(2, 0.35),
	crearBoton(3, 0.45),
	crearBoton(4, 0.55),
}

-- 游댳 Toggle 1: Mover Botones
local moverToggle = Instance.new("TextButton")
moverToggle.Size = UDim2.new(0, 120, 0, 40)
moverToggle.Position = UDim2.new(0.05, 0, 0.1, 0)
moverToggle.Text = "Mover: ON"
moverToggle.Font = Enum.Font.GothamBold
moverToggle.TextScaled = true
moverToggle.BackgroundColor3 = Color3.fromRGB(60, 60, 60)
moverToggle.TextColor3 = Color3.fromRGB(255, 255, 255)
moverToggle.Parent = gui

moverToggle.MouseButton1Click:Connect(function()
	moverActivo = not moverActivo
	moverToggle.Text = "Mover: " .. (moverActivo and "ON" or "OFF")
	for _, btn in ipairs(botones) do
		btn.Draggable = moverActivo
	end
end)

-- 游댳 Toggle 2: Ocultar Skills
local ocultarToggle = Instance.new("TextButton")
ocultarToggle.Size = UDim2.new(0, 160, 0, 40)
ocultarToggle.Position = UDim2.new(0.05, 0, 0.2, 0)
ocultarToggle.Text = "Ocultar Skills: OFF"
ocultarToggle.Font = Enum.Font.GothamBold
ocultarToggle.TextScaled = true
ocultarToggle.BackgroundColor3 = Color3.fromRGB(60, 60, 60)
ocultarToggle.TextColor3 = Color3.fromRGB(255, 255, 255)
ocultarToggle.Parent = gui

ocultarToggle.MouseButton1Click:Connect(function()
	ocultarActivo = not ocultarActivo
	ocultarToggle.Text = "Ocultar Skills: " .. (ocultarActivo and "ON" or "OFF")
	if skillBar then
		skillBar.Enabled = not ocultarActivo
	end
end)
